def add(k, v)
  k = header_to_env_key(k)
  case existing = @env[k]
  when nil
    @env[k] = v
  when String
    @env[k] = [existing, v]
  when Array
    existing << v
  end
end

def each
  return to_enum(:each) unless block_given?

  @env.each do |k, v|
    next unless k = env_to_header_key(k)
    yield k, v
  end
end
