def insert(entity)
  raise Exceptions::DuplicateEntityIdException, entity.id if exists?(entity.id)

  all << entity
  all_by_id[entity.id.to_s] = entity
  entity
end

def remove(entity)
  return unless exists?(entity.id)

  all.delete(entity)
  all_by_id.delete(entity.id.to_s)
end
