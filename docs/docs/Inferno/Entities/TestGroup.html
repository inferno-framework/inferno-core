def short_id(new_short_id = nil)
  return @short_id = new_short_id if new_short_id

  @short_id ||= begin
    prefix = parent.respond_to?(:short_id) ? "#{parent.short_id}." : ''
    suffix = parent ? (parent.groups.find_index(self) + 1).to_s : 'X'
    "#{prefix}#{suffix}"
  end
end
